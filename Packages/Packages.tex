                                % Personal environments
\newcounter{total}[subsection]
\setcounter{total}{0}

\newenvironment{definition}[2]
    {   
        \stepcounter{total}
        { \color{blue} Definition \thesubsection.\arabic{total}: #1 }

        \begin{adjustbox}{minipage=#2, right, vspace=0.1cm 0cm}
    }
    {
        \end{adjustbox}
    }

\newenvironment{wtheorem}[2]
    {   
        \stepcounter{total}
        { \color{red} Theorem \thesubsection.\arabic{total}: #1}

        \begin{adjustbox}{minipage=#2, right, vspace=0.1cm 0cm}
    }
    {
        \end{adjustbox}
    }

\newenvironment{ltheorem}[2]
    {   
        \stepcounter{total}
        { \color{red} Theorem \thesubsection.\arabic{total}: #1}

        \begin{enumerate}[label=(\alph*), leftmargin=#2, itemsep=0.1cm]
    }
    {
        \end{enumerate}
    }    

\newenvironment{corollary}[2]

        \begin{adjustbox}{minipage=#2, right, vspace=0.1cm 0cm}

        \end{adjustbox}
    }
